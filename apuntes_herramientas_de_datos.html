Listas, Sets, Tuplas y Diccionarios

Lista list — ordenada, mutable, permite duplicados

list_ = [1, 2, 3, 4]

list_.append(5)           # agrega al final: [1,2,3,4,5]
list_.insert(0, 0)        # inserta en posicion 0: [0,1,2,3,4,5]
list_.extend([6, 7])      # agrega multiples: [0,1,2,3,4,5,6,7]
list_.count(7)            # cuenta ocurrencias de 7
list_.index(5)            # posicion de 5
list_.remove(6)           # elimina primera ocurrencia de 6
valor = list_.pop()       # elimina y retorna el ultimo
len(list_)                # longitud
list_.clear()             # vacia la lista

Set set — NO ordenado, mutable, SIN duplicados

set_ = {1, 2, 3, 4, 3, 4, 5}  # {1, 2, 3, 4, 5}

set_.add(6)               # agrega elemento
set_.update({1, 2, 7})    # agrega multiples
set_.remove(6)            # elimina (error si no existe)
valor = set_.pop()        # elimina y retorna uno aleatorio
len(set_)                 # longitud
set_.clear()              # vacia el set

Tupla tuple — ordenada, INMUTABLE, permite duplicados

tuple_ = (1, 2, 3, 1)

tuple_.index(1)           # posicion de 1 -> 0
tuple_.count(1)           # cuenta ocurrencias -> 2
len(tuple_)               # longitud -> 4
tuple_[2]                 # acceso por indice -> 3

Funciones y Procedimientos

Procedimiento: NO retorna valores (sin return).

Funcion: Retorna 1 valor (con return).

# Procedimiento
def suma_procedimiento(valor_1: float, valor_2: float):
    """Procedimiento que muestra la suma de 2 valores."""
    print(f"resultado: {valor_1 + valor_2}")

# Funcion
def suma_funcion(valor_1: float, valor_2: float) -> float:
    """Funcion que retorna la suma de 2 valores."""
    return valor_1 + valor_2

# Uso
suma_procedimiento(valor_1=10, valor_2=5)
resultado = suma_funcion(10, 5)

class Persona:
    def __init__(self, nombre: str, apellido: str, edad: int):
        self.nombre = nombre
        self.apellido = apellido
        self.edad = edad

    def __str__(self) -> str:

Archivo INI

Archivo de configuracion con secciones entre corchetes y pares nombre=valor.

from configparser import ConfigParser

# ESCRITURA
def escribir_archivo(archivo: str):
    datos = {
        "pais_1": {
            "nombre": "Uruguay",
            "anio": 2008,
            "vecinos": [{"nombre": "Argentina", "direccion": "Oeste"},
                        {"nombre": "Brazil", "direccion": "Norte"}]
        }
    }
    ini = ConfigParser()
    ini.read_dict(datos)
    with open(archivo, "wt") as w:
        ini.write(w)

# LECTURA
def leer_archivo(archivo: str):
    ini = ConfigParser()
    ini.read(archivo)
    for seccion, valores in ini.items():
        if len(valores) > 0:
            print(f"[{seccion}]")
            for nombre, valor in valores.items():
                print(f"{nombre} = {valor}")

Archivo XML

Lenguaje de marcado extensible para almacenamiento y transmision de datos. Muy usado en APIs.

import xml.etree.ElementTree as ET
from xml.dom.minidom import parseString, parse

# ESCRITURA
def escribir_archivo(archivo: str):
    datos = ET.Element("datos")

    pais = ET.SubElement(datos, "pais", {"nombre": "Uruguay"})
    anio = ET.SubElement(pais, "anio")
    anio.text = "2008"
    vecinos = ET.SubElement(pais, "vecinos")
    ET.SubElement(vecinos, "vecino", {"nombre": "Argentina", "direccion": "Oeste"})

    with open(archivo, "wt") as w:
        w.write(parseString(ET.tostring(datos)).toprettyxml())

# LECTURA
def leer_archivo(archivo: str):
    dato = parse(archivo)
    print('\n'.join([line for line in dato.toprettyxml().split('\n') if line.strip()]))

Archivo CSV

Representacion de tablas en texto plano. Columnas separadas por comas (u otro separador).

from csv import reader, writer

# ESCRITURA
def escribir_archivo(archivo: str):
    datos = [
        ["col_1", "col_2", "col_3"],
        ["A", "B", "C"],
        ["D", "E", "F"],
    ]
    with open(archivo, "wt") as w:
        write_csv = writer(w)
        write_csv.writerows(datos)

# LECTURA
def leer_archivo(archivo: str):
    with open(archivo) as r:
        read_csv = reader(r)
        for line in read_csv:
            print(line)
Si un valor contiene comas, el modulo csv lo envuelve automaticamente entre comillas. Si escribis manualmente, tenes que manejar esto vos.

Archivo JSON

Notacion de objetos JavaScript. Alternativa a XML para intercambio de datos, muy usado en APIs web.

from json import dump, dumps, load

# ESCRITURA
def escribir_archivo(archivo: str):
    datos = {
        "datos": [{
            "pais": {"nombre": "Uruguay", "anio": 2008,
                     "vecinos": [{"nombre": "Argentina", "direccion": "Oeste"}]}
        }]
    }
    with open(archivo, "wt") as w:
        dump(datos, w, indent=4)  # indent para formato legible

# LECTURA
def leer_archivo(archivo: str):
    with open(archivo) as r:
        dato = load(r)
    print(dumps(dato, indent=4))  # dumps = dict -> string JSON

Funciones del modulo re

cadena = "La inteligencia artificial surge en 1940..."

# compile - compilar patron (recomendado si se usa muchas veces)
patron = re.compile(r"\d{4}")

# search - buscar primera coincidencia
span = re.search(r"\d", cadena).span()
# Retorna (inicio, fin) de la primera coincidencia

# findall - encontrar TODAS las coincidencias
re.findall(r"\d{4}", cadena)      # ['1940', '1950', ...]

# finditer - iterator de coincidencias (con info de posicion)
for match in re.finditer(r"\d{4}", cadena):
    print(match)

# split - dividir por patron
re.split(r"\s(el|la|los|las)\s", cadena)

# sub - reemplazar
re.sub(r"\d", "--", cadena)       # reemplaza digitos por "--"

# match - buscar al INICIO del string
re.match(r"La", cadena)           # Match si empieza con "La"

Ejemplos practicos de patrones

# Buscar anos (4 digitos)
patron = re.compile(r"\d{4}")
re.findall(patron, cadena)

# Buscar 1 o mas digitos
patron = re.compile(r"\d+")
re.findall(patron, cadena)

# Case-insensitive con flag
patron = re.compile(r"(?i)(el|l[ao]s?)\s")

# Reemplazar puntuacion
patron = re.compile(r"(\.|,|;)")
re.sub(patron, "#", cadena)

# Palabras terminadas en 'as' o 'os'
patron = re.compile(r"[ao]s\b")
re.sub(patron, "#", cadena)

Validar email

patron = re.compile(r"^\w+((\.|-|_)?\w+)*@\w+((\.|-)?\w+)*(\.\w{2,4})+$")
# ^\w+              = COMIENZA con 1+ letras
# ((\.|-|_)?\w+)*   = opcionalmente punto/guion/_ + letras (repetible)
# @                 = arroba obligatoria
# \w+               = 1+ letras del dominio
# ((\.|-)?\w+)*     = subdominio opcional
# (\.\w{2,4})+$     = TERMINA con .xx a .xxxx

email = "lcd-had141@ugr.edu.ar"
"OK!" if re.match(patron, email) else "ERROR"

Validar fecha (yyyy-mm-dd)

patron = re.compile(r"(\d{4}|\d{2})-(0[1-9]|1[0-2]|[1-9])-([1-9]|0[1-9]|[12]\d|3[01])$")
# (\d{4}|\d{2})             = ano 2 o 4 digitos
# (0[1-9]|1[0-2]|[1-9])     = mes 01-12 o 1-9
# ([1-9]|0[1-9]|[12]\d|3[01]) = dia 1-31

fecha = "1993-12-31"
"OK!" if re.match(patron, fecha) else "ERROR"
